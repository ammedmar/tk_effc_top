\newcommand{\icosahedron}[3][black]{\resizebox{#2}{#3}{
    \begin{tikzpicture}
      \tdplotsetmaincoords{0}{0}
      \tdplotsetrotatedcoords{13.6}{12.9}{-5.8}
      \begin{scope}[tdplot_rotated_coords,
        line width=1.5pt,line cap=round,line join=round,
        draw=#1]
        \pgfmathsetmacro{\phi}{(1 + sqrt(5))/2}
        \foreach \a/\b in {-1/-\phi,-1/\phi,1/-\phi,1/\phi} {
          \draw (0,\a,\b) -- (1,\a*\phi,0) -- (-1,\a*\phi,0) -- cycle;
          \draw (\a,\b,0) -- (\a*\phi,0,1) -- (\a*\phi,0,-1) -- cycle;
          \draw (\b,0,\a) -- (0,1,\a*\phi) -- (0,-1,\a*\phi) -- cycle;
        }
      \end{scope}
    \end{tikzpicture}
  }
}
